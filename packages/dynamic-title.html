# vuepress-plugin-dynamic-title

  • Website: www.sigure.xyz/vuepress-lovely-plugins/
  • MyBlog: www.sigure.xyz

# Install

yarn add vuepress-plugin-dynamic-title -D
# or use npm
npm i vuepress-plugin-dynamic-title -D

# Usage

module.exports = {
  plugins: ['dynamic-title']
}

# Options

module.exports = {
  plugins: [
    [
      "dynamic-title",
      {
        showIcon: "/favicon.ico",
        showText: "(/≧▽≦/)咦！又好了！",
        hideIcon: "/failure.ico",
        hideText: "(●—●)喔哟，崩溃啦！",
        recoverTime: 2000
      }
    ]
  ]
}
