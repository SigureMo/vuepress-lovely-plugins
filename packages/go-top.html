# vuepress-plugin-go-top

  • Website: www.sigure.xyz/vuepress-lovely-plugins/
  • MyBlog: www.sigure.xyz

# Install

yarn add vuepress-plugin-go-top -D
# or use npm
npm i vuepress-plugin-go-top -D

# Usage

// .vuepress/config.js
module.exports = {
  plugins: ['go-top']
}

# Responsive breakpoints

// .vuepress/styles/palette.styl

$MQWide ?= 1440px
$MQMobile ?= 768px
